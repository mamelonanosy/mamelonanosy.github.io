function editGroup(group, ref, editButton) { var newStrings = prompt("Please enter the new list of strings separated by newlines:"); if (newStrings != null) { var newStringsArray = newStrings.split('\n'); ref.set({}); newStringsArray.forEach(function(string, index) { ref.child(index.toString()).set(string); }); var linksDiv = group.getElementsByClassName("links")[0]; linksDiv.innerHTML = newStringsArray.map((value, index) => `${index}: ${value}`).join('
'); // Re-create the Edit button var newEditButton = document.createElement("button"); newEditButton.innerHTML = "Edit"; newEditButton.onclick = function() { editGroup(group, ref, newEditButton); }; // Check if the old Edit button is defined and still in the DOM if (editButton && editButton.parentNode) { group.replaceChild(newEditButton, editButton); } else { group.appendChild(newEditButton); } } }
